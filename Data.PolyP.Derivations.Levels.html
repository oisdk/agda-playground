levels : ⟦ FOREST ⟧ ~ A → ⟦ LEVELS ⟧ ~ A
levels t = go t []′
  where
  go : ⟦ FOREST ⟧ ~ A → ⟦ LEVELS ⟧ ~ A → ⟦ LEVELS ⟧ ~ A
  go =
    cata λ xs zs →
      cata (λ  { (inl _                             )  → zs
               ; (inr (inl _         ,  qs       )  )  → qs
               ; (inr (inr (x , xs)  ,  []′      )  )  → (x ∷′ []′  ) ∷′ xs []′
               ; (inr (inr (x , xs)  ,  q ∷′ qs  )  )  → (x ∷′ q    ) ∷′ xs qs
               }) xs
